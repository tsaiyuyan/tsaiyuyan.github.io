create.sh

#!/bin/bash
echo "create rootCA.key............"
openssl dsaparam -genkey -out rootCA.key 1024

echo "create rootCA.pem............"
openssl req -x509 -new -nodes -key rootCA.key -sha256 -days 1024 -out rootCA.pem -config rootca.conf

echo "create dh1024.pem............"
openssl dhparam -out dh1024.pem 1024

echo "create server.key............"
openssl ecparam -name prime256v1 -out s_prime256v1.pem
openssl ecparam -in s_prime256v1.pem -genkey -noout -out server.key

echo "create server_pub.pem............"
openssl ec -in server.key -pubout -out server_pub.pem

echo "create serverdsa.pem............"
openssl dsaparam -genkey -out serverdsa.pem 1024

echo "create server.csr............"
openssl req -new -key serverdsa.pem -out server.csr -config server.conf

echo "create server_cert.pem............"
openssl x509 -req -in server.csr -CAkey ./rootCA.key -CA ./rootCA.pem -force_pubkey server_pub.pem -out server_cert.pem -CAcreateserial

echo "convert to server.pem....."
cat  server_cert.pem server.key > server.pem

echo "clear files......"
rm -f rootCA.key  rootCA.srl  s_prime256v1.pem server.csr  server.key server_cert.pem  server_pub.pem  serverdsa.pem

echo "rootCA.pem server.pem dh1024.pem is needy."
